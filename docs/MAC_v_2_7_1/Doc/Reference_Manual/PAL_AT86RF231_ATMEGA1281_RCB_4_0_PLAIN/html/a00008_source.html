00060 void pal_trx_irq_init(FUNC_PTR trx_irq_cb)
00061 {
00062     /*
00063      * Set the handler function.
00064      * The handler is set before enabling the interrupt to prepare for spurious
00065      * interrupts, that can pop up the moment they are enabled
00066      */
00067     irq_hdl_trx = (irq_handler_t)trx_irq_cb;
