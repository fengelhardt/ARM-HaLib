00057 void pal_trx_read_timestamp(uint32_t *timestamp)
00058 {
00059     /*
00060      * Everytime a transceiver interrupt is triggred, input capture register of
00061      * the AVR is latched. The 'sys_time' is concatenated to the ICR to
00062      * generate the time stamp of the received frame.
00063      * 'sys_time'   'ICR'
00064      *  ---------|--------- => 32 bit timestamp
00065      *   16 bits   16 bits
00066      */
00067     *timestamp = (uint32_t)sys_time << (uint32_t)16;
00068     *timestamp |= (uint32_t)TIME_STAMP_REGISTER;
00069 }

00081 bool pal_calibrate_rc_osc(void)
00082 {
00083     /*
00084      * Use the 32.768 kHz external crystal oscillator as reference clock source.
00085      */
00086     /*
00087      * This is the actual value of the timer to be calibrated
00088      * after each calibration run.
00089      */
00090     uint16_t curr_value_cal_timer = 0;
00091 
00092     /* This is the best OSCCAL value. */
00093     uint8_t best_value_osccal = 0;
00094     /*
00095      * This is the smallest difference between the target value and the actual value
00096      * of the timer the timer to be calibrated.
00097      */
00098 
00099     uint16_t best_dif_timers = ~(uint16_t)0;
00100     /*
00101      * This is the difference between  between the target value and the actual value
00102      * of the timer the timer to be calibrated of the current calibration attempt.
00103      */
00104     uint16_t curr_dif_timers = 0;
00105 
00106     uint16_t counter;
00107     uint8_t tccr2b;
00108     uint8_t tccr1b;
00109     uint8_t tccr1a;
00110     bool cal_result = true;
00111 
00112     ENTER_CRITICAL_REGION();
00113 
00114     /*
00115      * Save current values of timer status.
00116      */
00117     tccr2b = TCCR2B;
00118     tccr1b = TCCR1B;
00119     tccr1a = TCCR1A;
00120 
00121     /*
00122      * Stop timers 1 and 2.
00123      * Set timer 1 to normal mode (no CTC, no PWM, just count).
00124      */
00125     TCCR2B = 0;
00126     TCCR1B = 0;
00127     TCCR1A = 0;
00128 
00129     for (counter = 0; counter < MAX_CAL_LOOP_CNT;  counter++)
00130     {
00131         /*
00132          * Delete pending timer 1 and 2 interrupts, and clear the
00133          * counters.
00134          */
00135         TIFR1 = 0xFF;
00136         TIFR2 = 0xFF;
00137         TCNT2 = 0;
00138         TCNT1 = 0;
00139 
00140         /* Kick timer1 with internal clock source and no prescaler */
00141         TCCR1B = (1 << CS10);
00142 
00143         /*
00144          * Kick timer2 with external 32.768 Hz asynchronous clock
00145          * and no prescaler
00146          */
00147         TCCR2B = (1 << CS20);
00148         ASSR = (1 << AS2);
00149 
00150         /*
00151          * Wait for timer 2 to overflow.
00152          */
00153         while (!(TIFR2 & (1 << TOV2)))
00154         {
00155             /* Wait */
00156         }
00157 
00158 
00159         /*
00160          * Stop timer 1.  Now, TCNT1 contains the number of CPU cycles
00161          * counted during F_CPU / (32 * 256) cycles.
00162          */
00163         TCCR1B = 0;
00164         TCCR2B = 0;
00165 
00166         curr_value_cal_timer = TCNT1;
00167 
00168         /*
00169          * Check if reference timer is running at all,
00170          * i.e. the reference clock is available.
00171          */
00172         if (0 == curr_value_cal_timer)
00173         {
00174             /* Reference timer is not running, return error. */
00175             cal_result = false;
00176             break;
00177         }
00178 
00179         if (curr_value_cal_timer <= (uint16_t)(TARGETVAL_CALIBRATION))
00180         {
00181             curr_dif_timers = (uint16_t)(TARGETVAL_CALIBRATION) - curr_value_cal_timer;
00182         }
00183         else
00184         {
00185             curr_dif_timers = curr_value_cal_timer - (uint16_t)(TARGETVAL_CALIBRATION);
00186         }
00187 
00188         if (curr_dif_timers < best_dif_timers)
00189         {
00190             best_dif_timers = curr_dif_timers;
00191             best_value_osccal = OSCCAL;
00192         }
00193 
00194         if (curr_value_cal_timer <= (uint16_t)(TARGETVAL_CALIBRATION))
00195         {
00196             /* Too slow, increase speed */
00197             OSCCAL++;
00198         }
00199         else
00200         {
00201             /* Too fast, lower speed */
00202             OSCCAL--;
00203         }
00204     }
00205 
00206     TCCR2B = tccr2b;
00207     TCCR1B = tccr1b;
00208     TCCR1A = tccr1a;
00209     LEAVE_CRITICAL_REGION();
00210 
00211     OSCCAL = best_value_osccal;
00212 
00213     return (cal_result);
00214 }

00283 void pal_led(led_id_t led_no, led_action_t led_setting)
00284 {
00285     uint8_t pin;
00286     /* New values of LED pins based on new LED state. */
00287     uint8_t led_pin_value;
00288     /*
00289      * Original value of LED port before writing new value.
00290      * This value needs to be restored.
00291      */
00292     uint8_t orig_led_port = LED_PORT & ~LED_BIT_MASK;
00293 
00294     /* Both LEDs need to be updated, since several peripherals
00295      * are dealing with the same port for this board
00296      * (USB, EEPROM, LEDs, Button).
00297      */
00298     LED_PORT_DIR |= (1 << LED_BIT_0);
00299     LED_PORT_DIR |= (1 << LED_BIT_1);
00300 
00301     switch (led_no)
00302     {
00303         default:
00304         case LED_0:
00305             pin = LED_BIT_0;
00306             break;
00307         case LED_1:
00308             pin = LED_BIT_1;
00309             break;
00310     }
00311 
00312     switch (led_setting)
00313     {
00314         case LED_ON:
00315             led_state |= _BV(pin);
00316             break;
00317 
00318         case LED_OFF:
00319             led_state &= ~_BV(pin);
00320             break;
00321 
00322         case LED_TOGGLE:
00323         default:
00324             if (led_state & _BV(pin))
00325             {
00326                 /*
00327                  * LED is currently on,
00328                  * Switch it off
00329                  */
00330                 led_state &= ~_BV(pin);
00331             }
00332             else
00333             {
00334                 /*
00335                  * LED is currently off,
00336                  * Switch it on
00337                  */
00338                 led_state |= _BV(pin);
00339             }
00340             break;
00341     }
00342 
00343     led_pin_value = (uint8_t)(~(uint16_t)led_state);  // Implicit casting required to avoid IAR Pa091.
00344     led_pin_value &= LED_BIT_MASK;
00345 
00346     LED_PORT = orig_led_port | led_pin_value;
00347 
00348     led_helper_func();
00349 }

00355 static void led_helper_func(void)
00356 {
00357     /*
00358      * Enable LED address decoding.
00359      * This is similar to USB, but with other settings.
00360      */
00361     LED_ADDR_DEC_PORT |= _BV(6);    // Different to USB
00362     LED_ADDR_DEC_DDR |= _BV(6);
00363     LED_ADDR_DEC_PORT &= ~_BV(7);
00364     LED_ADDR_DEC_DDR |= _BV(7);
00365 
00366     /* Enable desired LED state. */
00367     DDRG |= _BV(0);
00368     /* Set PE4 to 0 and back to 1. */
00369     PORTG &= ~_BV(0);
00370     PORTG |= _BV(0);
00371 
00372     /*
00373      * Disable LED address decoding.
00374      * This enables USB again.
00375      */
00376     LED_ADDR_DEC_PORT &= ~_BV(6);
00377     LED_ADDR_DEC_DDR |= _BV(6);
00378     LED_ADDR_DEC_PORT &= ~_BV(7);
00379     LED_ADDR_DEC_DDR |= _BV(7);
00380 
00381     /* Set the pins back to input */
00382     LED_PORT_DIR &= ~(LED_BIT_MASK);
00383 }

00422     BUTTON_PORT |= (1 << BUTTON_PIN_0);

00424

00426 
00427     PORTG |= _BV(1);
00428 
00429     /* Switch port back to output. */
00430     BUTTON_PORT_DIR |= (1 << BUTTON_PIN_0);
00431 
00432     /*
00433      * Disable button address decoding.
00434      * This enables USB again.
00435      */
00436     BUTTON_ADDR_DEC_PORT &= ~_BV(6);
00437     BUTTON_ADDR_DEC_DDR |= _BV(6);
00438     BUTTON_ADDR_DEC_PORT &= ~_BV(7);
00439     BUTTON_ADDR_DEC_DDR |= _BV(7);
00440 
00441     if (cur_button_state & 0x01)

00443         return BUTTON_PRESSED;
