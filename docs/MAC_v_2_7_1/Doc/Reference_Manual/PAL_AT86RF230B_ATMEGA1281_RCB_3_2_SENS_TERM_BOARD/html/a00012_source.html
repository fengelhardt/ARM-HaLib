00186 uint8_t sio_usb_tx(uint8_t *data, uint8_t length)
00187 {
00188     uint8_t bytes_to_be_written;
00189     uint8_t size;
00190     uint8_t back;
00191 
00192     /*
00193      * USB Tx has been marked as busy, e.g. maybe the terminal is not open.
00194      * In order to continue the execution of the application even if
00195      * USB Tx data cannot be delivered, we are going to return here
00196      * immediately.
00197      * Once USB Tx is not busy anymore, regular USB Tx procedures will
00198      * be resumed.
00199      */
00200     if (usb_tx_busy)
00201     {
00202         return (0);
00203     }
00204 
00205     /*
00206      * Calculate available buffer space
00207      */
00208     if (usb_tx_buf_tail >= usb_tx_buf_head)
00209     {
00210         size = (USB_TX_BUF_SIZE - 1) - (usb_tx_buf_tail - usb_tx_buf_head);
00211     }
00212     else
00213     {
00214         size = (usb_tx_buf_head - 1) - usb_tx_buf_tail;
00215     }
00216 
00217     if (size < length)
00218     {
00219         /* Not enough buffer space available. Use the remaining size */
00220         bytes_to_be_written = size;
00221     }
00222     else
00223     {
00224         bytes_to_be_written = length;
00225     }
00226 
00227     /* Remember the number of bytes transmitted. */
00228     back = bytes_to_be_written;
00229     usb_tx_count += bytes_to_be_written;
00230 
00231     /* The data is copied to the transmit buffer. */
00232     while (bytes_to_be_written > 0)
00233     {
00234         usb_tx_buf[usb_tx_buf_tail] = *data;
00235 
00236         if ((USB_TX_BUF_SIZE - 1) == usb_tx_buf_tail)
00237         {
00238             /* Reached the end of buffer, revert back to beginning of buffer. */
00239             usb_tx_buf_tail = 0;
00240         }
00241         else
00242         {
00243             /*
00244              * Increment the index to point the next character to be
00245              * inserted.
00246              */
00247             usb_tx_buf_tail++;
00248         }
00249 
00250         bytes_to_be_written--;
00251         data++;
00252     }
00253 
00254     /*
00255      * If a transmission needs to be started, pal_usb_handler() takes
00256      * care about that once it is run.
00257      */
00258 
00259     return back;
00260 }

00325 uint8_t sio_usb_rx(uint8_t *data, uint8_t max_length)
00326 {
00327     uint8_t data_received = 0;
00328 
00329     if (usb_rx_count == 0)
00330     {
00331         /* USB receive buffer is empty. */
00332         return 0;
00333     }
00334 
00335     if (USB_RX_BUF_SIZE <= usb_rx_count)
00336     {
00337          /*
00338           * Bytes between head and tail are overwritten by new data.
00339           * The oldest data in buffer is the one to which the tail is
00340           * pointing. So reading operation should start from the tail.
00341           */
00342         usb_rx_buf_head = usb_rx_buf_tail;
00343 
00344         /*
00345          * This is a buffer overflow case. Byt still only bytes equivalent to
00346          * full buffer size are useful.
00347          */
00348         usb_rx_count = USB_RX_BUF_SIZE;
00349 
00350         /* Bytes received is more than or equal to buffer. */
00351         if (USB_RX_BUF_SIZE <= max_length)
00352         {
00353              /*
00354               * Requested receive length (max_length) is more than the
00355               * max size of receive buffer, but at max the full
00356               * buffer can be read.
00357               */
00358              max_length = USB_RX_BUF_SIZE;
00359         }
00360     }
00361     else
00362     {
00363         /* Bytes received is less than receive buffer maximum length. */
00364         if (max_length > usb_rx_count)
00365         {
00366             /*
00367              * Requested receive length (max_length) is more than the data
00368              * present in receive buffer. Hence only the number of bytes
00369              * present in receive buffer are read.
00370              */
00371             max_length = usb_rx_count;
00372         }
00373     }
00374 
00375     data_received = max_length;
00376 
00377     while (max_length > 0)
00378     {
00379         /* Start to copy from head. */
00380         *data = usb_rx_buf[usb_rx_buf_head];
00381 
00382         if (++usb_rx_buf_head == USB_RX_BUF_SIZE)
00383         {
00384             usb_rx_buf_head = 0;
00385         }
00386 
00387         usb_rx_count--;
00388         data++;
00389         max_length--;
00390     }
00391     return data_received;
00392 }

00444 void usb_handler(void)
00445 {
00446     /*
00447      * USB Status is the upper 2 bits of Port E
00448      * PE6 - TXE  write is allowed if Low
00449      * PE7 - RXF  Data available if Low
00450      */
00451 
00452 #ifdef USB_FTDI_BITBANG
00453     uint8_t cUSBstat = USB_CTRL_PIN;
00454 #else
00455     uint8_t cUSBstat = USB_PIN;
00456 #endif  /* USB_FTDI_BITBANG */
00457 
00458     if (LOW == (cUSBstat & USB_RXF))
00459     {
00460         /* The received data is read in the PAL USB buffer */
00461         usb_getc();
00462     }
00463 
00464     /* Transmit data, if any */
00465     if (usb_tx_count > 0)
00466     {
00467         uint8_t bytes_tx = MAX_BYTES_FOR_USB_TX;
00468 
00469         while ((usb_tx_count > 0) && (bytes_tx > 0))
00470         {
00471             /*
00472              * To avoid slow transmission (byte per pal_task), a maximum of
00473              * MAX_BYTES_FOR_USB_TX bytes or number of bytes present in USB
00474              * buffer, whichever is minimum is transmitted at one go.
00475              */
00476             if (LOW == (cUSBstat & USB_TXE))
00477             {
00478                 if (usb_tx_busy)
00479                 {
00480                     /*
00481                      * Reset USB busy handling variables and mark USB as not busy.
00482                      */
00483                     usb_tx_busy = false;
00484                     usb_tx_busy_start_time = 0;
00485                 }
00486 
00487                 usb_putc();
00488 
00489                 bytes_tx--;
00490             }
00491             else
00492             {
00493                 /*
00494                  * In case USB Tx is already marked busy, we can skip the
00495                  * USB Tx busy detection algorithm.
00496                  */
00497                 if (!usb_tx_busy)
00498                 {
00499                     /*
00500                      * We have a USB Tx busy condition
00501                      * (else path to "LOW == (cUSBstat & USB_TXE)").
00502                      *
00503                      * So we need to check whether we may have the situation
00504                      * that we cannot deliver the data to the host,
00505                      * e.g. because the terminal program is not open.
00506                      * If this happens for a certain period of time, we have to
00507                      * suspend data transmission via USB. All data will be
00508                      * discarded.
00509                      *
00510                      * Once the USB Tx is not busy anymore, we will resume
00511                      * USB Tx handling.
00512                      */
00513                     if (usb_tx_busy_start_time == 0)
00514                     {
00515                         /*
00516                          * We are in the USB Tx busy condition the first time,
00517                          * so we need to read the current time to have an
00518                          * indication when busy condition starts.
00519                          */
00520                         pal_get_current_time(&usb_tx_busy_start_time);
00521                     }
00522                     else
00523                     {
00524                         uint32_t usb_tx_busy_current_time;
00525                         /*
00526                          * Check whether the USB tx busy condition is asserted too long.
00527                          */
00528                         pal_get_current_time(&usb_tx_busy_current_time);
00529 
00530                         if (SUB_TIME(usb_tx_busy_current_time, usb_tx_busy_start_time) > USB_TX_BUSY_TIME)
00531                         {
00532                             /*
00533                              * USB Tx is busy too long, stop all data delivery for now.
00534                              */
00535                              usb_tx_busy = true;
00536                         }
00537                     }
00538                 }   /* if (LOW == (cUSBstat & USB_TXE)) */
00539                 break;  /* while ((usb_tx_count > 0) && (bytes_tx > 0)) */
00540             }   /* while ((usb_tx_count > 0) && (bytes_tx > 0)) */
00541         }
00542     }   /* if (usb_tx_count > 0) */
00543 }
